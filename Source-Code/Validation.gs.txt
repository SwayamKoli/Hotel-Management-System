/**
 * ============================================================
 * Validation Functions
 * ============================================================
 */

function isBlank(value) {
  return value === "" || value === null;
}

function isValidMobile(mobile) {
  return /^[6-9]\d{9}$/.test(mobile);
}

function isValidEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}
